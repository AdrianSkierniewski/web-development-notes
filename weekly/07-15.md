    class Person extends Eloquent
    {
        public function search()
        {
            $items = self::select(array('id'));
            return Datatables::of($items)
                ->add_column('name', '{{ MyModel::find($id)->name }}', 0)
                ->remove_column('id')
                ->make();
        }

        {
            return $this->first_name . ' ' . $this->last_name;
